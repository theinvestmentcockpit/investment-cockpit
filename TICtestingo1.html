from flask import Flask, request, jsonify import sqlite3 import pandas as pd app = Flask(__name__) # Database connection def get_db_connection(): conn = sqlite3.connect('stocks.db') conn.row_factory = sqlite3.Row return conn # Route to fetch all stocks @app.route('/api/stocks', methods=['GET']) def get_stocks(): conn = get_db_connection() stocks = conn.execute('SELECT * FROM stocks').fetchall() conn.close() return jsonify([dict(stock) for stock in stocks]) # Route to apply formula-based filters @app.route('/api/filters', methods=['POST']) def apply_filters(): filters = request.json.get('filters') conn = get_db_connection() df = pd.read_sql_query('SELECT * FROM stocks', conn) conn.close() try: result = df.query(filters) return result.to_json(orient='records') except Exception as e: return jsonify({'error': str(e)}), 400 # Sample route to fetch dummy news sentiment @app.route('/api/news', methods=['GET']) def get_news(): news_sample = [ {"headline": "Market hits new high", "sentiment": "positive"}, {"headline": "Stock crash in tech sector", "sentiment": "negative"} ] return jsonify(news_sample) if __name__ == '__main__': app.run(debug=True)
